let loadMoreBtn = document.createElement('button');
loadMoreBtn.innerText = "Load More Posts";
document.querySelector('main').appendChild(loadMoreBtn);

loadMoreBtn.addEventListener('click', function() {
    let newPost = document.createElement('article');
    newPost.classList.add('post');
    newPost.innerHTML = `
        <h2>New Blog Post</h2>
        <p>Author: Jane Doe | Date: March 15, 2025</p>
        <p>This is a new blog post that will be shown when you click the Load More button.</p>
        <a href="#">Read more</a>
    `;
    document.querySelector('main').appendChild(newPost);
});
